
            // Function to check if third-party API keys are set up
            function checkThirdPartyApiKeys() {
                fetch('/api/key-status')
                    .then(response => {
                        // Always try to parse the response, even if status isn't 200
                        return response.json().catch(() => {
                            // If we can't parse the response, assume no keys
                            return { has_keys: false, error: 'Failed to parse response' };
                        });
                    })
                    .then(data => {
                        console.log("API key status:", data);
                        // Handle case when API returns has_keys: false or there's an error
                        if (!data.has_keys) {
                            // Remove any existing notification first
                            const existingNotification = document.querySelector('.third-party-api-notification');
                            if (existingNotification) {
                                existingNotification.remove();
                            }
                            
                            // Create and show the API key notification
                            const apiKeyNotification = document.createElement('div');
                            apiKeyNotification.className = 'third-party-api-notification';
                            apiKeyNotification.innerHTML = `
                                <div class="third-party-api-notification-content">
                                    <i data-lucide="key"></i>
                                    <span>You need to set up your third-party API keys to use AI features. <a href="/settings">Go to Settings</a> to add your keys. Features won't function as expected.</span>
                                    <button class="third-party-api-notification-close">×</button>
                                </div>
                            `;
                            document.body.appendChild(apiKeyNotification);
